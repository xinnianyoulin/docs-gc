      # 使用

      引用全部组件

      import gridComponents from "@/assets/grid-components.min.js";

Vue.use(gridComponents);

      引用部分组件

      import { GridCells, ... , } from '@/assets/grid-components.min.js'

Vue.use(GridCells).use(...)

或者在组件components项中注入

export default {
  ...
  components: { GridCells, ...},
  ...
}

      使用

      <!-- 在html模块中直接使用 -->
<grid-cells></grid-cells>

<grid-card></grid-card>

<data-table />
...

      # 树形项

      提示

      默认 url 中返回的数据格式为数组，每个项之间有相互父子关联字段（pid/id）
      通过treeOptions来自动格式化

      // 树形项
const config = {
  // 必传，过滤控件类型
  type: "tree",
  // 必传，筛选项label
  label: "",
  // 必传，事件中返回值的key{[key]:value}
  key: "",
  //数据集合，url不为空时，以url为主
  data: [],
  //数据接口地址
  url: "",
  // 树形关联字段及子级字段map
  treeOptions: {
    // 顶级入口pid
    entry: 0,
    // 需要的树形层级
    level: null,
    // 树形关联项字段
    maps: {
      id: "id",
      pid: "pid",
      children: "children"
    }
  },
  // 控件字段map 使用的label及value
  keyMaps: {
    // 显示字段 *树形项默认label为name*
    label: "name",
    // 最终使用的参数值字段 *树形项默认value为id*
    value: "id",
    // 子集字段与上述treeOptions.maps.children 相应
    children: "children"
  },
  //默认选择值
  default: ""
};

const config = {

  type: "select",

  label: "",

      自定义值转时间区间特殊说明

      下拉项中自定义值转区间，实现业务上几小时或者几天前快速转时间区间
      data 内部子项的 valueToRange 权限高于外部，针对当前项的 value 转换成最近时间

      /*
 * data内部子项的valueToRange权限高于外部，针对当前项的value转换成最近时间
 */
const valueToRange = {
  type: "select",
  label: "最近时间",
  key: "time", // 最终返回字段
  valueToRange: "days", // 可选值[hours,days]
  format: "yyyy-MM-dd 08:00:00",
  data: [
    {
      label: "最近一小时",
      valueToRange: "hours", // 可选值[hours,days]
      value: "1"
    },
    {
      name: "最近七天",
      value: "7"
    }
  ]
};

      # 时间项

      提示：时间关键字 nowTime

      时间项的 default 内部设定了关键字 nowTime
      nowTime：生成当前时间
      nowTime-subHours-7： 生成当前时间 7 小时前
      nowTime-subDays-7：生成当前时间 7 天前
      特别的，在datarange对象中default字段支持[nowTime-subDays-7, nowTime]来生成默认请求时间

      const config = {
  // 必传，过滤控件类型 [year,month,date,datetime]
  type: "",
  // 必传，筛选项label
  label: "",
  // 必传，主配置中所使用的接口字段
  key: "",
  // 时间值返回格式化 年：yyyy,月：yyyy-MM，日：yyyy-MM-dd，时间：yyyy-MM-dd HH:mm:ss
  valueFormat: "yyyy-MM-dd HH:mm:ss",
  // 时间项显示格式化 同上
  format: "yyyy-MM-dd HH:mm:ss",
  //默认选择值  可接收时间字符串 2018-01-01 08:00:00 或date对象
  default: "" // nowTime 关键字
};

      # 时间区间项

      // 时间区间项
// range内部 开始时间与结束时间配置与时间项各类型相同
const config = {
  type: "daterange", // 只允许有一个
  key: "time",
  // daterange的valueFormat，在range内部无效，format有效
  valueFormat: "yyyy-MM-dd HH:mm:ss",
  range: [
    {
      type: "datetime",
      label: "开始时间",
      format: "yyyy-MM-dd HH:mm:ss"
    },
    {
      type: "datetime",
      label: "结束时间",
      default: "2018-11-12 08:00:00"
    }
  ],
  default: "[nowTime-subDays-7,nowTime]" // default 同样支持nowTime组合
};

      # 级联项

      提示

      groups数组内每一项配置与普通项配置相同，当前可选input, tree, select, autoComplete

      // 级联项
const config = {
  type: "group",
  // 最终使用的参数key 与关联项中使用的一致
  key: "value",
  groups: [
    {
      type: "autocomplete",
      label: "站点名称",
      key: "name",
      bridge: {
        to: "level2" // 级联关联对象
      },
      fetchParam: "name1",
      url: "demo.api.address",
      keyMaps: {
        label: "name",
        value: "code"
      }
    },
    {
      type: "select",
      label: "关联项",
      key: "type2", // 唯一标识，需要存在
      bridge: {
        to: "level3",
        from: "level2"
      },
      url: "demo.api.address",
      keyMaps: {
        label: "name",
        value: "code"
      }
    },
    {
      type: "select",
      label: "关联项",
      key: "ret3", // 最终合并到的参数key
      bridge: {
        from: "level3"
      },
      url: "demo.api.address",
      keyMaps: {
        label: "name",
        value: "code"
      }
    }
  ]
};
